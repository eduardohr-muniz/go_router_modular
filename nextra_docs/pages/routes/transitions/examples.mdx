# Transition Examples

This page provides practical examples of how to implement page transitions in Go Router Modular applications.

## Basic Transition Examples

### Fade Transitions

```dart
class FadeExampleModule extends Module {
  @override
  List<ModularRoute> get routes => [
    ModuleRoute(
      '/fade',
      module: FadeExampleModule(),
      transition: GoTransitions.fade,
      duration: Duration(milliseconds: 300),
    ),
  ];
}
```

## Inheritance Examples

### Parent Module with Transition

```dart
class ParentModule extends Module {
  @override
  List<ModularRoute> get routes => [
    ModuleRoute(
      '/parent',
      module: ParentModule(),
      transition: GoTransitions.fadeUpwards,
      duration: Duration(milliseconds: 300),
    ),
  ];
}
```

### Child Routes Inheriting and Overriding

```dart
class ChildModule extends Module {
  @override
  List<ModularRoute> get routes => [
    // Inherits fadeUpwards from parent
    ChildRoute('/', child: (_, __) => HomePage()),
    
    // Overrides with slide transition
    ChildRoute('/details', 
      child: (_, __) => DetailsPage(),
      transition: GoTransitions.slide.toLeft,
      duration: Duration(milliseconds: 400),
    ),
    
    // Overrides with scale transition
    ChildRoute('/profile', 
      child: (_, __) => ProfilePage(),
      transition: GoTransitions.scale.withFade,
      duration: Duration(milliseconds: 500),
    ),
  ];
}
```

## Global Configuration Example

### App-Wide Transition Settings

```dart
class AppWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    // Configure global defaults
    GoTransition.defaultCurve = Curves.easeInOut;
    GoTransition.defaultDuration = Duration(milliseconds: 400);

    return MaterialApp.router(
      routerConfig: Modular.routerConfig,
      title: 'Transition Example App',
      theme: ThemeData(
        // Platform-specific fallbacks
        pageTransitionsTheme: PageTransitionsTheme(
          builders: {
            TargetPlatform.android: GoTransitions.fadeUpwards,
            TargetPlatform.iOS: GoTransitions.cupertino,
            TargetPlatform.macOS: GoTransitions.cupertino,
            TargetPlatform.linux: GoTransitions.fade,
            TargetPlatform.windows: GoTransitions.fade,
            TargetPlatform.fuchsia: GoTransitions.fade,
          },
        ),
      ),
    );
  }
}
```

## Complete App Example

### Multi-Module App with Different Transitions

```dart
class AppModule extends Module {
  @override
  List<ModularRoute> get routes => [
    // Home with fade transition
    ModuleRoute(
      '/',
      module: HomeModule(),
      transition: GoTransitions.fadeUpwards,
      duration: Duration(milliseconds: 300),
    ),
    
    // Products with slide transition
    ModuleRoute(
      '/products',
      module: ProductsModule(),
      transition: GoTransitions.slide.toRight.withFade,
      duration: Duration(milliseconds: 400),
    ),
    
    // Profile with scale transition
    ModuleRoute(
      '/profile',
      module: ProfileModule(),
      transition: GoTransitions.scale.withFade,
      duration: Duration(milliseconds: 500),
    ),
    
    // Settings with rotate transition
    ModuleRoute(
      '/settings',
      module: SettingsModule(),
      transition: GoTransitions.rotate.withFade,
      duration: Duration(milliseconds: 600),
    ),
  ];
}
```

  child: (_, __) => HomePage(),
